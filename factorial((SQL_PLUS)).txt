*CODE*:

SET SERVEROUTPUT ON;
SET VERIFY OFF;
Declare
num integer;
factorial integer := 1;
begin
num := &num;
if num < 0 then
raise_application_error(-20001,'factorial is not defined for negative numbers');
else
for i in 1..num loop
factorial := factorial * i;
end loop;
dbms_output.put_line('The factorial of ' || num || ' is ' || factorial);
end if;
exception
when others then
dbms_output.put_line('Error : ' || sqlerrm);
end;
/
